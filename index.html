//setup code
      const SIZE = 600;
      ARC_SPACING = 3;
      SPEED = 1 / 450000.0
      function setup() {
        createCanvas(SIZE, SIZE);
        background(0)
      }
      //draw pattern every frame
      function draw() { drawArcs(); }
      
      function drawArcs() {
        //starting at the edge of the screen, draw an arc then move inward
        for (arcS = SIZE; arcS > 1; arcS -= ARC_SPACING) {
          //this magic color defines the gradient, a faded edge to the pattern, and the pattern slowly emerging at the start
          let fadedEdge = sqrt(sqrt(min(millis() * 100, SIZE * 0.9))) + min(millis() / 100, SIZE * 0.9) - (arcS * arcS / SIZE);
          fill(color((arcS * 100 / SIZE) + 250, 300, fadedEdge));

          //This code draws the arcs! 
          angle = (arcS) * (millis() * PI * SPEED)
          arc(SIZE / 2, SIZE / 2, arcS, arcS, -angle, angle - PI);
        }
      }
