   size = 400
   speed = 1/100
   spacing = 4
   function setup() {
     colorMode(HSB)
     createCanvas(size, size);
     background(0)
   }
   //draw pattern every frame
   function draw() {
     //starting at the edge of the screen, draw an arc then move inward
     for (radius = size; radius > 1; radius -= spacing) {
       fill((radius * 120 / size) + 240, 75, 100, alpha);
       //defines and draws arc from time and arc distance from center
       angle = (size - radius) * millis() * speed / (spacing / 2);
       arc(size / 2, size / 2, radius, radius, -angle * 2, angle);
     }
   }
